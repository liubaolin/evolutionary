排序过滤器 | orderBy

字符串

{{ [{name:'richey',age:18}, {name:'richey',age:16}, {name:'richey',age:17}, {name:'java',age:10}, {name:'pythen',age:20} ] | orderBy:'name' }}

函数

{{ [{name:'richey',age:18}, {name:'richey',age:16}, {name:'richey',age:17}, {name:'java',age:10}, {name:'pythen',age:20} ] | orderBy:ordersFunc }}

数组

{{ [{name:'richey',age:18}, {name:'richey',age:16}, {name:'richey',age:17}, {name:'java',age:10}, {name:'pythen',age:20} ] | orderBy:['name','-age'] }}
